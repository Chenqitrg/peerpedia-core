EXAMPLES
  # Search by partial name
  peerpedia account search Alice

  # Search case-insensitive
  peerpedia account search alice     # same result as Alice

  # Search with a fragment
  peerpedia account search Chen      # matches "Chenqi Meng", "Chen Li", ...

  # JSON output (machine-readable)
  peerpedia account search Alice --json
  # → [{"id": "abc12345-...", "name": "Alice Wang"}, ...]
